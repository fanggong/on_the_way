
### 5.3 数据库抽检

```sql
-- 账号与角色
select email, profile_completed, status from app.user_account order by created_at desc limit 5;
select u.email, r.role_code
from app.rbac_user_role r
join app.user_account u on u.user_id = r.user_id
order by u.created_at desc;

-- 系统核心源与同步策略
select * from app.system_core_source order by system_code;

-- 同步任务
select system_code, connector_code, job_type, status, triggered_at, started_at, finished_at
from app.connector_sync_job
order by triggered_at desc
limit 20;
```
